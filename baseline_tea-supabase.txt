CREATE TABLE baseline_tea (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    paciente_id VARCHAR(10) UNIQUE NOT NULL,
    data_triagem DATE NOT NULL,
    nome_paciente VARCHAR(255) NOT NULL,
    data_nascimento DATE NOT NULL,
    idade_paciente INTEGER,
    cuidador_nome VARCHAR(255) NOT NULL,
    cuidador_parentesco VARCHAR(100),
    cuidador_profissao VARCHAR(255),
    residencia TEXT,
    queixa_principal TEXT NOT NULL,
    idade_diagnostico_tea INTEGER NOT NULL,
    nivel_suporte INTEGER NOT NULL,
    profissional_diagnostico VARCHAR(255) NOT NULL,
    outros_diagnosticos TEXT,
    historico_gestacao TEXT,
    tipo_parto VARCHAR(50),
    idade_gestacional INTEGER,
    condicoes_nascer TEXT,
    marco_cabeca TEXT,
    marco_locomocao TEXT,
    desenvolvimento_fala TEXT,
    controle_esfincteriano TEXT,
    meltdowns_frequencia TEXT,
    meltdowns_duracao TEXT,
    meltdowns_gatilhos TEXT,
    agressividade_autoagressao TEXT,
    estereotipias TEXT,
    hipersensibilidade TEXT,
    rigidez_cognitiva TEXT,
    contato_visual TEXT,
    padrao_sono TEXT,
    alimentacao_seletividade TEXT,
    funcao_intestinal TEXT,
    convulsoes_eeg TEXT,
    alergias TEXT,
    imunidade TEXT,
    medicacoes_atuais TEXT,
    medicacoes_previas TEXT,
    terapias_atuais TEXT,
    objetivo_terapia_canabinoide TEXT NOT NULL,
    experiencia_cannabis_previa TEXT,
    consentimento_assinado BOOLEAN NOT NULL,
    assinatura_responsavel VARCHAR(255),
    data_consentimento DATE,
    data_criacao TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    data_atualizacao TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

CREATE INDEX idx_baseline_paciente_id ON baseline_tea(paciente_id);
